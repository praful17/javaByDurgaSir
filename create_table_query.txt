CREATE TABLE DIV_LOOKUP
(
ID NUMERIC PRIMARY KEY,
LABEL VARCHAR(255) UNIQUE,
CODE VARCHAR(10)
);

CREATE TABLE SUBDIV_LOOKUP
(
ID NUMERIC PRIMARY KEY,
LABEL VARCHAR(255) UNIQUE,
CODE VARCHAR(10),
FK_DIV_CODE VARCHAR(10)
);

ALTER TABLE SUBDIV_LOOKUP ADD FOREIGN KEY (FK_DIV_CODE) REFERENCES DIV_LOOKUP(CODE);

CREATE TABLE SESCA_CLASSIFICATION_LOOKUP
(
ID NUMERIC PRIMARY KEY,
LABEL VARCHAR(255) UNIQUE,
CODE VARCHAR(10)
);

CREATE TABLE CLASSIFICATION_LOOKUP
(
ID NUMERIC PRIMARY KEY,
LABEL VARCHAR(255) UNIQUE,
CODE VARCHAR(10),
FK_SUBDIV_CODE VARCHAR(10),
FK_SESCA_CLASSIFICATION_CODE VARCHAR(10)
);

ALTER TABLE CLASSIFICATION_LOOKUP ADD FOREIGN KEY (FK_SUBDIV_CODE) REFERENCES SUBDIV_LOOKUP(CODE);
ALTER TABLE CLASSIFICATION_LOOKUP ADD FOREIGN KEY (FK_SESCA_CLASSIFICATION_CODE) REFERENCES SESCA_CLASSIFICATION_LOOKUP(CODE);
